beta1MCO = function( x, y ){
  S(x,y)/S(x,x)
}

beta0MCO = function( x, y ){
  mean(y) - beta1MCO(x, y) * mean(x)
}


########### Question 4) ##############

beta1RI = function( x, y ){
  S(y,y)/S(x,y)
}

beta0RI = function( x, y ){
  mean(y) - beta1RI(x, y) * mean(x)
}
